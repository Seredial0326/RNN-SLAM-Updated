# CMakeLists.txt

cmake_minimum_required(VERSION 3.20)
project(RNNSLAM)

# C++ Standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Build type
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

# Compiler flags
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -march=native")
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3")
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -g")

# Find packages
find_package(Eigen3 3.4 REQUIRED)
find_package(OpenCV 4 REQUIRED)
find_package(Pangolin REQUIRED)
find_package(Boost REQUIRED COMPONENTS system thread filesystem)

# CUDA (opcional, pero recomendado)
find_package(CUDA)
if(CUDA_FOUND)
    message(STATUS "CUDA found: ${CUDA_VERSION}")
    include_directories(${CUDA_INCLUDE_DIRS})
    add_definitions(-DUSE_CUDA)
endif()

# Include directories
include_directories(
    ${PROJECT_SOURCE_DIR}/src
    ${PROJECT_SOURCE_DIR}/thirdparty/dso/src
    ${EIGEN3_INCLUDE_DIR}
    ${OpenCV_INCLUDE_DIRS}
    ${Pangolin_INCLUDE_DIRS}
)

# DSO library
add_subdirectory(thirdparty/dso)

# Source files
set(RNNSLAM_SOURCES
    src/main.cpp
    src/RNNDepthEstimator.cpp
    src/DataConverter.cpp
)

# Main executable
add_executable(rnn_slam ${RNNSLAM_SOURCES})

# Link libraries
target_link_libraries(rnn_slam
    ${OpenCV_LIBS}
    ${Pangolin_LIBRARIES}
    ${Boost_LIBRARIES}
    dso
)

if(CUDA_FOUND)
    target_link_libraries(rnn_slam ${CUDA_LIBRARIES})
endif()

# Installation
install(TARGETS rnn_slam DESTINATION bin)
